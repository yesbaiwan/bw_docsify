# 七合一聚合

七个渠道每个都建一个 worker 太麻烦了，所以做了一个聚合 worker，将七个渠道的 API 都聚合到一个 worker 中。

七个渠道的名称是：

- `kilo`
- `longcat`
- `aiping`
- `ling`
- `poe`
- `openrouter`
- `qiniuyun`

每个渠道的通过 `/渠道名` 来访问，例如，你的 worker 地址是 `https://xxx.workers.dev/`，那么 `/kilo` 就是 `https://xxx.workers.dev/kilo`，推荐自定义域名使用。

代码里还留了一个小彩蛋，快去看看吧！

```js
const CHANNELS = {
  kilo: {
    base: "https://api.kilo.ai/api/openrouter",
    root: "Hello World",
    async models() {
      const res = await fetch(`${this.base}/models`);
      if (!res.ok) return { data: [], object: "list" };
      const json = await res.json();
      const free = json.data?.filter((m) => m.isFree === true) || [];
      return { object: json.object, data: free };
    },
    transform(path) {
      return path.replace(/^\/v1/, "");
    },
  },

  longcat: {
    base: "https://api.longcat.chat/openai",
    root: "模型列表请查看 https://longcat.chat/platform/docs/zh/#支持的模型",
    async models() {
      const res = await fetch("https://longcat.chat/platform/docs/zh/");
      const html = await res.text();
      const models = [];
      for (const row of html.matchAll(/<tr>[\s\S]*?<\/tr>/gi)) {
        const cell = row[0].match(/<t[dh]>([\s\S]*?)<\/t[dh]>/i)?.[1];
        const name = cell?.replace(/<.*?>/g, "").trim();
        if (name?.includes("LongCat")) models.push({ id: name, object: "model" });
      }
      return { data: models, object: "list" };
    },
  },

  aiping: {
    base: "https://aiping.cn/api",
    root: "请在 https://aiping.cn/user/called-records 查看调用记录",
    async models() {
      const res = await fetch("https://aiping.cn/api/v1/models");
      if (!res.ok) return { data: [], object: "list" };
      const { data, object } = await res.json();
      const free = data.filter((m) => m.price?.input_price_range?.[0] === 0 && m.price?.output_price_range?.[0] === 0);
      return { object, data: free };
    },
  },

  ling: {
    base: "https://api.tbox.cn/api/llm",
    root: "获取密钥前往 https://ling.tbox.cn/open",
    async models() {
      try {
        const res = await fetch("https://lingstudio.tbox.cn/meta/model/list", {
          headers: { referer: "https://ling.tbox.cn/" },
        });
        if (!res.ok) return { data: [], object: "list" };
        const json = await res.json();
        const data = (json.data || []).filter((m) => m.status === "ACTIVE").map((m) => ({ id: m.code, object: "model" }));
        return { data, object: "list" };
      } catch {
        return { data: [], object: "list" };
      }
    },
  },

  poe: {
    base: "https://api.poe.com",
    root: "获取密钥前往 https://poe.com/api/keys",
    async models(req) {
      const res = await fetch(`${this.base}/v1/models`, req);
      if (!res.ok) return { data: [], object: "list" };
      const json = await res.json();
      const free = json.data?.filter((m) => m.pricing?.request === "0.00") || [];
      return { object: json.object, data: free };
    },
  },

  gogoapi: {
    base: "https://api.gogoapi.top",
    root: "获取密钥前往 https://api.gogoapi.top",
    async models() {
      const res = await fetch(`${this.base}/api/pricing`);
      if (!res.ok) return { data: [], object: "list" };
      const json = await res.json();
      const free = json.data?.filter((m) => m.model_ratio === 0 && m.model_price === 0) || [];
      const data = free.map((m) => ({ id: m.model_name, object: "model" }));
      return { object: "list", data };
    },
  },

  openrouter: {
    base: "https://openrouter.ai/api",
    root: "获取密钥前往 https://openrouter.ai/keys",
    async models(req) {
      const res = await fetch(`${this.base}/v1/models`, req);
      if (!res.ok) return { data: [], object: "list" };
      const json = await res.json();
      json.data = json.data?.filter((m) => m.pricing?.prompt === "0" && m.pricing?.completion === "0") || [];
      return json;
    },
  },

  qiniuyun: {
    base: "https://api.qnaigc.com",
    root: "模型列表请查看 https://www.qiniu.com/ai/models",
    async models() {
      const res = await fetch("https://www.qiniu.com/ai/models");
      const html = await res.text();
      const start = html.indexOf('"models"', html.indexOf('"models"') + 1);
      const section = html.slice(start);
      const models = [];
      for (const match of section.matchAll(/"id":\s*"([^"]+)"[\s\S]*?"retirement_at":\s*"([^"]*)"/gi)) {
        const id = match[1];
        if (id && !(match[2] && new Date(match[2]) < new Date())) {
          models.push({ id, object: "model" });
        }
      }
      return { data: models, object: "list" };
    },
  },
};

async function handleChannel(channel, path, search, req) {
  const config = CHANNELS[channel];
  if (!config) {
    return new Response("Unknown channel", { status: 404 });
  }

  if (path === "/" || path === "") {
    return new Response(config.root, { status: 200 });
  }

  if (path === "/models" || path === "/v1/models") {
    const result = await config.models(req);
    return Response.json(result);
  }

  const upstreamPath = config.transform ? config.transform(path) : path;
  return fetch(`${config.base}${upstreamPath}${search}`, req);
}

export default {
  async fetch(req) {
    const url = new URL(req.url);
    const pathParts = url.pathname.split("/").filter(Boolean);

    if (pathParts.length === 0) {
      return new Response("API Gateway - Available channels: /kilo, /longcat, /aiping, /ling, /poe, /openrouter, /gogoapi, /qiniuyun", { status: 200 });
    }

    const channel = pathParts[0];
    const subPath = "/" + pathParts.slice(1).join("/");

    return handleChannel(channel, subPath, url.search, req);
  },
};
```
